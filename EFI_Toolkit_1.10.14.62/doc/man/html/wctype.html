NAME

     iswalnum, iswalpha, iswcntrl, iswdigit, iswgraph, iswlower,
     iswprint, iswpunct, iswspace, iswupper, iswxdigit, tolower, toupper, -
     wide character classification macros

SYNOPSIS

     #include <wctype.h>

     int
     iswalnum(wchar_t wc)

     int
     iswalpha(wchar_t wc)

     int
     iswcntrl(wchar_t wc)

     int
     iswdigit(wchar_t wc)

     int
     iswgraph(wchar_t wc)

     int
     iswlower(wchar_t wc)

     int
     iswprint(wchar_t wc)

     int
     iswpunct(wchar_t wc)

     int
     iswspace(wchar_t wc)

     int
     iswupper(wchar_t wc)

     int
     iswxdigit(wchar_t wc)

     wchar_t
     towlower(wchar_t wc)

     wchar_t
     towupper(wchar_t wc)

     The above functions perform character tests and conversions on the wide
     char wc, for US English only. They are available as macros, defined in 
     the include file <wctype.h>.	See the specific manual pages 
     for more information.
